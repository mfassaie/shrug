    ```rust
    /// Module-scoped prefix prevents collisions with e2e/auth.rs unique_name()
    /// which uses the same process ID. The "sc" prefix is for static_commands.
    fn unique_name(prefix: &str) -> String {
        format!("sc-{}-{}", prefix, std::process::id())
    }

    fn create_test_profile(runner: &SmokeRunner, name: &str) { ... }
    fn delete_test_profile(runner: &SmokeRunner, name: &str) { ... }

    /// Panic-safe cleanup guard. Calls delete_test_profile on Drop,
    /// so profiles are cleaned up even if an assertion panics mid-test.
    struct ProfileGuard<'a> {
        runner: &'a SmokeRunner,
        names: Vec<String>,
    }
    impl<'a> ProfileGuard<'a> {
        fn new(runner: &'a SmokeRunner) -> Self {
            Self { runner, names: vec![] }
        }
        fn track(&mut self, name: String) {
            self.names.push(name);
        }
    }
    impl<'a> Drop for ProfileGuard<'a> {
        fn drop(&mut self) {
            for name in &self.names {
                delete_test_profile(self.runner, name);
            }
        }
    }
    ```
